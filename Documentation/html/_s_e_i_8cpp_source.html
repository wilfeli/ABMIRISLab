343 SEI::form_design_for_params(std::shared_ptr<PVProject> project_, double demand, double solar_irradiation, double permit_difficulty, double project_percentage, const IterTypeDecSM& iter, PVDesign& design)
344 {
345  design.solar_irradiation = solar_irradiation;
346 
347  design.permit_difficulty = permit_difficulty;
348 
349  design.N_PANELS = std::ceil(demand * project_percentage / ((solar_irradiation) * iter.second->efficiency * (iter.second->length * iter.second->width/1000000) * ( 1 - WorldSettings::instance().params_exog[EParamTypes::DCtoACLoss])));
350 
351  design.PV_module = iter.second;
352 
353  design.DC_size = design.N_PANELS * iter.second->STC_power_rating / constants::NUMBER_WATTS_IN_KILOWATT;
354 
355  design.AC_size = design.DC_size * (1 - WorldSettings::instance().params_exog[EParamTypes::DCtoACLoss]);
356 
358 
359 
360 
361  design.hard_costs = design.N_PANELS * iter.second->efficiency * THETA_hard_costs[0] + std::pow(design.DC_size, 2) * THETA_hard_costs[1];
362 
363  design.soft_costs = design.N_PANELS * THETA_soft_costs[0] + permit_difficulty * THETA_soft_costs[1];
364 
365  design.total_costs = (design.hard_costs + design.soft_costs) * THETA_profit[0];
366

378 SEI::ac_estimate_savings(PVDesign& design, std::shared_ptr<PVProject> project_)
379 {
380  //estimate savings for each project
381  //get price of kW from the utility, assume increase due to inflation
382  //total life-time of a project
383 
384 
385 
388 
389  //simple calculation when HH owns the system
390  auto inflation = WorldSettings::instance().params_exog[EParamTypes::InflationRate];
391  auto CPI = 1;
392  auto energy_costs = 0.0;
393  auto AC_size = design.AC_size;
394  auto degradation_t = std::exp(std::log((1 - design.PV_module->degradation))/WorldSettings::instance().params_exog[EParamTypes::DegradationDefinitionLength]);
395  for (auto i = 0; i < design.PV_module->warranty_length; ++i)
396  {
397  //estimate yearly energy production, if PPA might be used in estimation
398  //MARK: cont. check again
399  auto yearly_production = AC_size * design.solar_irradiation * constants::NUMBER_DAYS_IN_YEAR;
400  energy_costs += yearly_production * CPI * WorldSettings::instance().params_exog[EParamTypes::ElectricityPriceUCDemand];
401  AC_size = AC_size * degradation_t;
402  CPI = CPI * inflation;
403  };
404 
405  design.energy_savings_money = energy_costs;
406  design.total_savings = energy_costs - design.total_costs;
