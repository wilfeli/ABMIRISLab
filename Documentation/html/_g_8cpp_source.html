152  //go through projects, if permit has been requested and processing time has elapsed - schedule a visit

154  for (auto& project:pending_pvprojects)
155  {
156  if (project->state_project == EParamTypes::RequestedPermitForInstallation)
157  {
158  auto project_permitting_time = params[EParamTypes::GProcessingTimeRequiredForGrantingPermitForInstallation] * w->world_map->g_map[project->agent->location_y][project->agent->location_x]->permit_difficulty;
159  if ((a_time - project->ac_g_time) >= project_permitting_time)

164  };
165 
166  };
167 
168 
169 
170  if (project->state_project == EParamTypes::RequestedInspectionAfterInstallation)
171  {
172  //if permit was requested - check that processing time after request has elapsed and contact agent to schedule visit, check capacity for visits for each future time
173  auto project_scheduling_time = params[EParamTypes::GProcessingTimeRequiredForSchedulingPermitVisit] * w->world_map->g_map[project->agent->location_y][project->agent->location_x]->permit_difficulty;
174  if ((a_time - project->ac_g_time) >= project_scheduling_time)
175  {
176  bool FLAG_SCHEDULED_VISIT = false;
177  std::size_t i_offset;
178  std::size_t i;
179  std::weak_ptr<PVProject> w_project = project;
180  while (!FLAG_SCHEDULED_VISIT && i_offset < schedule_visits.size())
181  {
182  //check that there is space for the visit
183  i = (i_schedule_visits + i_offset) % schedule_visits.size();
184 
185  if (schedule_visits[i].size() < params[EParamTypes::GMaxNVisitsPerTimeUnit])
186  {
187  auto agent_reply = project->agent->request_time_slot_visit(a_time + i_offset, w_project);
188 
189  if (agent_reply)
190  {
191  FLAG_SCHEDULED_VISIT = project->agent->schedule_visit(a_time + i_offset, w_project);
192 
193  if (FLAG_SCHEDULED_VISIT)
194  {
195  schedule_visits[i].push_back(w_project);
196  project->state_project = EParamTypes::ScheduledPermitVisit;
197  project->ac_g_time = a_time;
198  }
199  };
200  i_offset++;
201  };
202  };
203  };
204  };

206  {
207  //if permit was requested - check that processing time after request has elapsed
208  auto project_processing_time = params[EParamTypes::GProcessingTimeRequiredForProcessingPermit] * w->world_map->g_map[project->agent->location_y][project->agent->location_x]->permit_difficulty;
209  if ((a_time - project->ac_g_time) >= project_processing_time)
210  {

212  };
213  };
214  };
215

223  collect_inf_site_visit(project);
