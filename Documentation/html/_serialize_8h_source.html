462  template <class T>
463  T solve_str_formula(const std::string& formula_, IRandom& rand_)
464  {
465  T dest_;
466  std::string formula = formula_;
467 
468  if (formula.find("FORMULA::") != std::string::npos)
469  {
470  std::regex e("");
471  e.assign("FORMULA\\u003A\\u003A");
472  formula = std::regex_replace(formula, e, "");
473 
474  //check if it is requesting random number generation
475  if (formula.find("p.d.f.") != std::string::npos)
476  {
477  e.assign("p.d.f.\\u003A\\u003A");
478  formula = std::regex_replace(formula, e, "");
479 
480  //parse formula
481  //case of a truncated normal
482  if (formula.find("N_trunc") != std::string::npos)
483  {
484  double mean = std::stod(formula.substr(formula.find("(") + 1, formula.find(",") - formula.find("(") - 1));
485  double sigma2 = std::stod(formula.substr(formula.find(",") + 1, formula.find(",") - formula.find(")") - 1));
486 
488  formula = std::to_string(std::max(rand_.rnd() * std::pow(sigma2, 0.5) + mean, 0.0));
489 
490  }
492  else if (formula.find("u") != std::string::npos)
493  {
494  double min = std::stod(formula.substr(formula.find("(") + 1, formula.find(",") - formula.find("(") - 1));
495  double max = std::stod(formula.substr(formula.find(",") + 1, formula.find(",") - formula.find(")") - 1));
496 
497  formula = std::to_string(rand_.ru() * (max - min) + min);
498  };
499 
500  };
501 
502  }
503  else
504  {
505  };
506 
507  //convert string into value
508  DeserializeValue<T>::deserialize_value(std::to_string(solve_formula(formula)), dest_);
509 
510  return dest_;
511  }
