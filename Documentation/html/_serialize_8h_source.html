452  template <class T>
453  T solve_str_formula(const std::string& formula_, IRandom& rand_)
454  {
455  T dest_;
456  std::string formula = formula_;
457 
458  if (formula.find("FORMULA::") != std::string::npos)
459  {
460  std::regex e("");
461  e.assign("FORMULA\\u003A\\u003A");
462  formula = std::regex_replace(formula, e, "");
463 
464  //check if it is requesting random number generation
465  if (formula.find("p.d.f.") != std::string::npos)
466  {
467  e.assign("p.d.f.\\u003A\\u003A");
468  formula = std::regex_replace(formula, e, "");
469 
470  //parse formula
471  //case of a truncated normal
472  if (formula.find("N_trunc") != std::string::npos)
473  {
474  double mean = std::stod(formula.substr(formula.find("(") + 1, formula.find(",") - formula.find("(") - 1));
475  double sigma2 = std::stod(formula.substr(formula.find(",") + 1, formula.find(",") - formula.find(")") - 1));
476 
478  formula = std::to_string(std::max(rand_.rnd() * std::pow(sigma2, 0.5) + mean, 0.0));
479 
480  }
482  else if (formula.find("u") != std::string::npos)
483  {
484  double min = std::stod(formula.substr(formula.find("(") + 1, formula.find(",") - formula.find("(") - 1));
485  double max = std::stod(formula.substr(formula.find(",") + 1, formula.find(",") - formula.find(")") - 1));
486 
487  formula = std::to_string(rand_.ru() * (max - min) + min);
488  };
489 
490  };
491 
492  }
493  else
494  {
495  };
496 
497  //convert string into value
498  DeserializeValue<T>::deserialize_value(std::to_string(solve_formula(formula)), dest_);
499 
500  return dest_;
501  }
