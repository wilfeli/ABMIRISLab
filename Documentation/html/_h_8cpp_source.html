99 void
100 Household::ac_inf_quoting_sei()
101 {
102 
103  //move timer here
104  if ((!get_inf_marketing_sei.empty()) || quote_stage_timer > 0.0)
105  {
106  ++quote_stage_timer;
107  };
108 
109  //requests quotes from SEI
110  //restricts number of projects
111  while ((!get_inf_marketing_sei.empty()) && (pvprojects.size() <= WorldSettings::instance().constraints[EConstraintParams::MaxNOpenProjectsHH]))
112  {
113  auto marketing_inf = get_inf_marketing_sei.front();
114 
115  //create project
116  //create new project
117  auto new_project = std::make_shared<PVProject>();
118  //request additional information
119  new_project->agent = this;
120  new_project->begin_time = a_time;
121  new_project->sei = marketing_inf->agent;
122  //save project
123  pvprojects.push_back(new_project);
124  marketing_inf->agent->get_project(new_project);
125 
126  //Distinguishes between online and phone quote. Small installers might not have an online presence.
128  switch (marketing_state)
129  {
130  case EParamTypes::HHMarketingStateHighlyInterested:
131  //requests preliminary quote with site visit
132  new_project->state_project = EParamTypes::RequestedPreliminaryQuote;
133  marketing_inf->agent->request_preliminary_quote(new_project);
134  case EParamTypes::HHMarketingStateInterested:
135  switch (marketing_inf->sei_type)
136  {
137  case EParamTypes::SEISmall:
138  //returns quote in the form of a message
139  // assume no online presence, so requests preliminary quote
140  new_project->state_project = EParamTypes::RequestedPreliminaryQuote;
141  marketing_inf->agent->request_preliminary_quote(new_project);
142  break;
143  case EParamTypes::SEILarge:
144  //requests online quote, it will be provided in a separate call
145  new_project->state_project = EParamTypes::RequestedOnlineQuote;
146  marketing_inf->agent->request_online_quote(new_project);
147  break;
148 
149  default:
150  break;
151  }
152 
153  default:
154  //otherwise do nothing as is not interested
155  break;
156  };
157 
158  get_inf_marketing_sei.pop_front();
159  };

164 Household::dec_evaluate_online_quotes()
165 {
166  //sort projects by price, lower price goes first
167  std::sort(pvprojects.begin(), pvprojects.end(), [](const std::shared_ptr<PVProject> lhs, const std::shared_ptr<PVProject> rhs){
168  //compare only if online quote was received,
169  bool compare_res = false;
170  if (lhs->state_project == EParamTypes::ProvidedOnlineQuote && rhs->state_project == EParamTypes::ProvidedOnlineQuote)
171  {
172  compare_res = lhs->online_quote->params[EParamTypes::OnlineQuotePrice] < rhs->online_quote->params[EParamTypes::OnlineQuotePrice];
173  };
174  return compare_res;
175  });
176 
177 
178  //mark first N projects for requesting preliminary quotes and request them
179  auto n_request_quotes = 0;
180  auto max_n_request_quotes = WorldSettings::instance().constraints[EConstraintParams::MaxNRequestedPreliminaryFromOnlineQuotes];
181  while (n_request_quotes < max_n_request_quotes)
182  {
183  auto project = pvprojects[n_request_quotes];
184  project->state_project = EParamTypes::RequestedPreliminaryQuote;
185  project->sei->request_preliminary_quote(project);
186  ++n_request_quotes;
187  };

194 void
195 Household::dec_evaluate_preliminary_quotes()
196 {
197  //sort projects by price, lower price goes first
198  std::sort(pvprojects.begin(), pvprojects.end(), [](const std::shared_ptr<PVProject> lhs, const std::shared_ptr<PVProject> rhs){
199  //compare only if online quote was received,
200  bool compare_res = false;
201  if (lhs->state_project == EParamTypes::ProvidedPreliminaryQuote && rhs->state_project == EParamTypes::ProvidedPreliminaryQuote)
202  {
203  compare_res = lhs->online_quote->params[EParamTypes::PreliminaryQuoteEstimatedSavings] > rhs->online_quote->params[EParamTypes::PreliminaryQuoteEstimatedSavings];
204  };
205  return compare_res;
206  });
207 
208 
209  std::shared_ptr<PVProject> decision = nullptr;
210  //pick top project and compare price to income
211  for(auto& project:pvprojects)
212  {
213  if (project->state_project == EParamTypes::ProvidedPreliminaryQuote)
214  {
215  if (project->preliminary_quote->params[EParamTypes::PreliminaryQuotePrice] <= params[EParamTypes::Income] * THETA_design[EParamTypes::HHDecPreliminaryQuote][0])
216  {
217  decision = project;
218  break;
219  };
220  }
221  };
222 
223  if (decision)
224  {
225  decision->state_project = EParamTypes::AcceptedPreliminaryQuote;
226  decision->sei->accepted_preliminary_quote(decision);
227  };
