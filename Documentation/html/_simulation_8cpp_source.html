205 
206  //calculate scale factor

208 
209  //draw first from conditional distribution
210  //create cmf - vector of end cumulative values from cond_dist
211  auto cmf = create_cmf(cond_dist, scale_factor, dist);
212 
213  //if it is discrete - just get inverse index, cmf has value and index; and bin value from that
214  //draw next uniform
215  u_i = rand->ru();
216  auto i = get_inverse_index(cmf, u_i);
217  i_x.push_back(dist.cond_values[cond_dist[i]].back());
218  if (dist.type == ERandomParams::Discrete)

220  x.push_back(dist.bin_values[i_x.back()]);
221  }
222  else if (dist.type == ERandomParams::ContinuousDiscretized)
223  {
224  //if it is continuous - get inverse index and inverse value from the corresponding description
225  //will point to the bin to use - simple linear inverse
226  //check if it is last bin and if it will include inf right end
227 
228  if (( i == dist.theta_bins.size() - 1) && (dist.bin_ends.back() == constants::SOLAR_INFINITY()))
229  {
230  x.push_back(get_inverse_value_exp(dist.theta_bins[i], u_i));

293 tools::create_cmf(std::vector<long>& cond_dist, double scale_factor, EmpiricalUVD& dist)
294 {
295  std::vector<double> cmf{0};
296  for (auto i = 0; i < cond_dist.size(); ++i)
297  {
298  cmf.push_back(cmf[i] + dist.cond_freq[cond_dist[i]] / scale_factor);
299  };
300 
301 
302  dist.theta_bins.clear();
303  //update theta_bins
304  for (auto i = 1; i < cmf.size(); ++i)
305  {
306  dist.theta_bins.push_back(std::vector<double>{});
307  if (( i != cmf.size() - 1) || ((i == cmf.size() - 1) && (dist.bin_ends.back() != constants::SOLAR_INFINITY())))
308  {
309  dist.theta_bins.back().push_back(cmf[i-1]);
310  dist.theta_bins.back().push_back(dist.bin_ends[i] - dist.bin_ends[i-1]);
311  dist.theta_bins.back().push_back(cmf[i] - cmf[i-1]);
312  dist.theta_bins.back().push_back(dist.bin_ends[i-1]);
313  }
314  else
315  {
316  dist.theta_bins.back().push_back(std::exp(dist.bin_ends[i-1]) * (1 - cmf[i-1]));
317  };
318  };
319 
320 
321  return cmf;
